import tensorflow as tf
from tensorflow.keras import datasets as tfds

(X_train,y_train),_ = tfds.mnist.load_data()

X_train = X_train.reshape(-1,28*28)
X_train = X_train / 255.0

data = tf.data.Dataset.from_tensor_slices((X_train.astype('float32'),y_train.astype('int32')))
data = data.batch(32,drop_remainder=True)
data = data.cache()
print(data)

## train a logistic regression

w = tf.Variable(tf.random.normal(shape=(28*28,10),stddev=0.1),trainable=True)
optimizer = tf.optimizers.SGD(0.01)

@tf.function
def train_step(x, y):
  with tf.GradientTape() as tape:
    all_logits = tf.matmul(x,w) # (n_batch,n_class)
    y_logits = tf.gather(all_logits,y,batch_dims=1) # (n_batch,)
    logp = y_logits - tf.reduce_logsumexp(all_logits,axis=1)
    loss = -logp
  gradients = tape.gradient(loss,[w])
  optimizer.apply_gradients(zip(gradients,[w]))

# wrap the code to be profiled with <start> and <end> methods
tf.profiler.experimental.start('logs/profile')
for (x,y) in data:
  train_step(x,y) 
tf.profiler.experimental.stop()
